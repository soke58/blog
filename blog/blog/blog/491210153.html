这个简单的VB程序只是一个外壳。通过内嵌IE的方式实现打开网站的功能。代码如下：

Dim form_w '定义变量，用于记录调整前长与宽

Dim form_h

Private Sub Command1_Click()

Dim IE As String

Dim url As String

If Text1.Text = "" Then MsgBox "请输入关键词"

Exit Sub

IE = Text1

WebBrowser2.Visible = True

WebBrowser1.Width = Form1.Width / 2

WebBrowser2.Width = Form1.Width / 2

If WebBrowser1.Width > Form1.Width / 2 Then WebBrowser1.Width = Form1.Width / 2

  WebBrowser1.Navigate "http://www.baidu.com/s?wd=" + IE + ""

  WebBrowser2.Navigate "http://www.google.com.hk/#hl=zh-CN&source=hp&q=" + IE + ""

End Sub

 

Private Sub Command2_Click()

      End

End Sub

 

Private Sub Command3_Click()

WebBrowser2.Visible = False

WebBrowser1.Width = Form1.Width

If IsNumeric(Text1.Text) Then

WebBrowser1.Navigate "http://www.qqxoo.com/main.html?qqid=" & Text1.Text

Else

MsgBox ("请输入正确的QQ号！"), 26, "警告" '设定弹出窗口

End If

End Sub

 

Private Sub Form_Load()

Dim IE, ID As String

ID = Text1.Text

IE = Text1

form_w = Form1.Width '记录初始长宽

form_h = Form1.Height

End Sub

 

Private Sub Form_Resize()

'On Error Resume Next

bl_W = Form1.Width / form_w '计算调整比例

bl_h = Form1.Height / form_h

 

 

WebBrowser1.Width = WebBrowser1.Width * bl_W '与上同理

WebBrowser1.Left = WebBrowser1.Left * bl_W

WebBrowser1.Height = WebBrowser1.Height * bl_h

WebBrowser1.Top = WebBrowser1.Top * bl_W

WebBrowser2.Width = WebBrowser2.Width * bl_W '与上同理

WebBrowser2.Left = WebBrowser2.Left * bl_W

WebBrowser2.Height = WebBrowser2.Height * bl_h

WebBrowser2.Top = WebBrowser2.Top * bl_W

 

form_w = Form1.Width '记录当前窗体尺寸,用于下一次调整时计算比例用.

form_h = Form1.Height

End Sub

 

Private Sub Text1_GotFocus()

Text1.SelStart = 0

Text1.SelLength = Len(Text1.Text) '过程调用

End Sub
