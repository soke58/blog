将长字符串插入分割后再拆分成数组

Dim str
Dim strSplit
Dim strRet
str =Trim(Request("key")) ''此处获得表单传来的值或者是变量值
str=Replace(str,"的","") ''此处获得表单传来的值或者是变量值中的字符替换为空
strSplit = ","

For j = 1 To Len(str)
strRet = strRet & Mid(str, j, 2) & strSplit
Next
if Len(strRet)>=3 then
strRet = Left(strRet, Len(strRet) - 3)
end if
if Len(strRet)=1 then
strRet = Left(strRet, Len(strRet) - 0)
end if
if Len(strRet)=2 then
strRet = Left(strRet, Len(strRet) - 2)
end if
response.write strRet

str=strRet ''此处获得分割后的数组再次拆分成若干数组
