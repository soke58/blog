一个ASP判断是不是输入值数字的方法

dim str                                                                                                           【  定义STR】
str=trim(request.form("baobei"))                                                                     【从name=baobei的表单提取输入的值宝贝baobei】
for   i=1   to   len(str)
    if   asc(mid(str,i,1))> 57   or   asc(mid(str,i,1)) <48   then                           【通过判断ASCII编码判断是不是数字】
          response.write"<script language='javascript'>alert('对不起!!宝贝数目必须是数字!');"        【弹出提示】
          response.write"window.history.go(-1);</script>"                                     【浏览器返回刚才输入的表单】
          exit   for
    end   if
next
