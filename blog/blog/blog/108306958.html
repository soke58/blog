一段调用数据库数据的简单代码

<%
strcon=``Provider=Microsoft.Jet.OLEDB.4.0;Data Source=``+Server.MapPath(``数据库.mdb``)
Set db=Server.createobject(``ADODB.connection``)
db.open(strcon)
set rs=server.createobject(``adodb.recordset``)
strSql=``select * from 表名``
rs.open strSql,db,1,1
%>
<table>
<%
do while not rs.eof
%>
<tr><td>
<%=rs(``字段1``)%>
</td>
</tr>
<tr><td>
<%=rs(``字段2``)%>
</td></tr>
<%
rs.movenext
if rs.eof then exit do
loop
set rs=nothing
%>
</table>
